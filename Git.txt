Общие принципы при заливки нового проекта
git init #инициализация гит в папке с проектом
git branch -M main #создаем ветку в папке с проектом куда будет сохранятся наш текущий проект
git remote add origin https://github.com/.....  #добавляем в нашей папке с проектом сведения удаленного репозитария с названием origin
git add -A #индуксируем все файлы локально
git commit -m "message сообщение коммита" #создаем коммит изменения
git push -u origin main #переносим локальные файлы из ветки main на удаленный репозит origin
